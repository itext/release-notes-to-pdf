using System;

using System.Collections.Generic;

using System.IO;

using iText.Kernel.Pdf;

using iText.Pdfocr;

using iText.Pdfocr.Onnx;

using iText.Pdfocr.Onnx.Detection;

using iText.Pdfocr.Onnx.Recognition;


namespace iText.Samples.Sandbox.Pdfocr.Onnx {

    /// <summary>PdfOcrOnnxPaddleOcrExample.cs</summary>

    /// <remarks>

    /// This example demonstrates how to perform OCR using <see cref="iText.Pdfocr.Onnx.OnnxOcrEngine"/>

    /// and PaddleOCR ML-models for the given list of input images and save output to a PDF file using provided path.

    /// PaddleOCR models converted to ONNX format can be found at

    /// <a href="https://huggingface.co/itextresearch">iText Research Hugging Face page</a>.

    /// Required software: iText 9.6.0, pdfOCR-Onnx 5.0.0

    /// (itext.pdfocr.onnx.cpu dependency to execute ONNX models on CPU or

    /// itext.pdfocr.onnx.abstract and Microsoft.ML.OnnxRuntime.Gpu dependencies to execute ONNX models on GPU).

    /// </remarks>

    public class PdfOcrOnnxPaddleOcrExample {

        private const String DEST = "results/sandbox/pdfocr/onnx/PdfOcrOnnxPaddleOcrExample/result.pdf";

        private const String IMAGE = "../../../resources/img/image.png";

        private const String MODELS = "../../../resources/models/";

        private const String DETECTION = MODELS + "paddleocr/PP-OCRv5_mobile_det_infer";

        private const String RECOGNITION = MODELS + "paddleocr/PP-OCRv5_mobile_rec_infer";


        public static void Main(String[] args) {

            // It's possible to create custom IOrtSessionOptionsCreator and use it to create predictors.

            // By default, sessionOptionsCreator can be omitted.

            IOrtSessionOptionsCreator sessionOptionsCreator = new DefaultOrtSessionOptionsCreator();

            IDetectionPredictor detectionPredictor = OnnxDetectionPredictor.PaddleOcr(

                DETECTION + "/inference.onnx", DETECTION + "/inference.yml", sessionOptionsCreator);

            IRecognitionPredictor recognitionPredictor = OnnxRecognitionPredictor.PaddleOcr(

                RECOGNITION + "/inference.onnx", RECOGNITION + "/inference.yml", sessionOptionsCreator);

            using (OnnxOcrEngine ocrEngine = new OnnxOcrEngine(detectionPredictor, recognitionPredictor)) {

                OcrPdfCreator pdfCreator = new OcrPdfCreator(ocrEngine);

                pdfCreator.CreatePdf(new List<FileInfo> { new FileInfo(IMAGE) }, new PdfWriter(DEST)).Close();

            }

        }

    }

}
